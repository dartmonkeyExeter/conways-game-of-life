FROM MINECAFT IMPORT FUNCTION isPlayerWithinDistanceToVillage()

SET PlayerX TO 0
SET PlayerZ TO 0


PROCEDURE Find_Village(PlayerX, PlayerZ)
BEGIN PROCEDURE
	SET Search TO True

	WHILE Search DO
	BEGIN WHILE
		PlayerX = PlayerX + 1
		PlayerZ = PlayerZ + 1
		IF FUNCTION isPlayerWithinDistanceToVillage() THEN
			SET Search TO False
		END IF
	END WHILE
END PROCEDURE

PROCEDURE Golem()
BEGIN PROCEDURE
	SEND MINE 3 BLOCKS TO KEYBOARD
	SEND PUNCH GOLEM TO KEYBOARD
	SEND BUILD 3 BLOCK TOWER TO KEYBOARD
	SEND KILL GOLEM TO KEYBOARD
	SEND CRAFT BUCKET TO KEYBOARD
	SEND BREAK GRAVEL TO KEYBOARD
	SEND GET FLINT TO KEYBOARD
	SEND CRAFT FLINT AND STEEL TO KEYBOARD
	BEGIN WHILE

END PROCEDURE

PROCEDURE Find_Lava(PlayerX, PlayerZ)
BEGIN PROCEDURE
	SET Search TO True

	WHILE Search DO
		PlayerX = PlayerX + 1
		PlayerZ = PlayerZ + 1
		RECEIVE Lava_Nearby FROM KEYBOARD
		IF Lava_Nearby IS TRUE THEN
			SEND CREATE NETHERPORTAL WITH WATER AND LAVA TO KEYBOARD
			SEND LIGHT PORTAL TO KEYBOARD
			SEARCH = False
		END IF
	END WHILE
END PROCEDURE

Find_Village()
Golem()
Find_Lava()
